public class ByRef
{
    static <T> ByRef<T> Create(T value)
    {
        return new ByRef<T>(value);
    }

    static <T> ByRef<T> Create()
    {
        return new ByRef<T>();
    }
}

public class ByRef<T> : ByRef
{
    public T value;

    ByRef<T>() { }

    ByRef<T>(T value)
    {
        this.value = value;
    }
}
